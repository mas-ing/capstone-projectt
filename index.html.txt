  function getTopRecommendations(energy, emotion, vibe, topN = 3) {
    if (!energy || !emotion) {
      return musicLibrary.slice(0, topN);
    }
    
    const scored = musicLibrary.map(track => {
      let score = 0;
      const tags = track.moodTags;
      
      
      if (tags.emotion === emotion) score += 40;
      
      
      if (tags.energy === energy) score += 30;
      else if (tags.energy === 'medium' && energy === 'low') score += 8;
      else if (tags.energy === 'high' && energy === 'medium') score += 5;
      
     
      if (vibe && tags.vibe === vibe) score += 25;
      else if (vibe && tags.vibe && (vibe === 'upbeat' && tags.vibe === 'heartfelt')) score += 8;
      
      
      score += Math.random() * 0.1;
      
      return { track, score };
    });
    
    
    scored.sort((a, b) => b.score - a.score);
    return scored.slice(0, topN).map(item => item.track);
  }

        <a href="${song.link}" class="listen-link" target="_blank" rel="noopener noreferrer">
          🎧 Listen on YouTube ▶
        </a>
      </div>
    `).join('');
    
    const countBadge = document.getElementById('songCountBadge');
    if (countBadge) countBadge.innerText = `${songs.length} songs`;
  }
  
  
  function escapeHtml(str) {
    if (!str) return '';
    return str.replace(/[&<>]/g, function(m) {
      if (m === '&') return '&amp;';
      if (m === '<') return '&lt;';
      if (m === '>') return '&gt;';
      return m;
    });
  }
